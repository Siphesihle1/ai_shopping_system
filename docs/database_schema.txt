Table Customer as C {
  id int [pk, increment]
  user varchar
  cellphone varchar
  latitude varchar
  longitude varchar

}

Table Product as P {
  id int [pk, increment]
  price float
  name varchar
  digital boolean
  image image
}

Table Order as O {
	id int [pk,increment]
	customer int
	transaction_id varchar
	date_ordered DateTime
	complete boolean
}
Ref: O.id < C.customer

Table OrderItem as OI {
	id int [pk,increment]
	product int
	order int
	quantity int
	date_added DateTime
	
}
Ref: OI.id < P.product
Ref: OI.id < O.order

Table ShippingAddress as SA {

	id int [pk,increment]
	customer int 
	order int
	address varchar
	city varchar
	state varchar
	zipcode varchar
	date_added DateTime
	
}
Ref: SA.id < C.customer
Ref: SA.id < O.order
